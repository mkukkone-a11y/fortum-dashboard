function fetchUMMRSS() { const controller = new AbortController(); const timeoutId = setTimeout(() => controller.abort(), 10000); var now = new Date(); // 1. Pyöristetään päivät keskiyöhön, jotta osoite pysyy samana koko päivän var startDate = new Date(now.getFullYear(), now.getMonth(), now.getDate() - 7); var stopDate = new Date(now.getFullYear(), now.getMonth(), now.getDate() + 14); // 2. Otetaan päivät ulos puhtaana ISO-muotona (ILMAN ylimääräistä tuplakoodausta!) var startIso = startDate.toISOString(); var stopIso = stopDate.toISOString(); // 3. Rakennetaan puhdas alkuperäinen osoite var dynamicUrl = `https://ummrss.nordpoolgroup.com/messages/?messageTypes=5&areas=10YFI-1--------U&companies=N00290&searchText=river&publicationStartDate=2020-01-01T00:00:00.000Z&eventStartDate=${startIso}&eventStopDate=${stopIso}&limit=100`; // 4. Tehdään koodaus turvalliseen muotoon VAIN TÄSSÄ KERRAN välityspalvelimelle const proxyUrl = 'https://api.rss2json.com/v1/api.json?rss_url=' + encodeURIComponent(dynamicUrl); fetch(proxyUrl, { signal: controller.signal }) .then(r => r.json()) .then(data => { clearTimeout(timeoutId); if (data.status !== 'ok') { throw new Error(data.message || 'Palvelin hylkäsi pyynnön.'); } var items = data.items; var next7Days = new Date(now); next7Days.setDate(now.getDate() + 7); var processedEvents = []; for (var i = 0; i < items.length; i++) { var descHtml = items[i].content || items[i].description; var tempDiv = document.createElement('div'); tempDiv.innerHTML = descHtml; var rows = tempDiv.getElementsByTagName('tr'); var evtData = {}; for (var r = 0; r < rows.length; r++) { var th = rows[r].getElementsByTagName('th')[0]; var td = rows[r].getElementsByTagName('td')[0]; if (th && td) { var key = th.innerText.replace(':', '').trim().toLowerCase(); evtData[key] = td.innerText.trim(); } } var eventStart = parseNordPoolDate(evtData['event start']); var eventStop = parseNordPoolDate(evtData['event stop']); if (eventStart && eventStart <= next7Days && (!eventStop || eventStop > now)) { var titleText = evtData['unit name'] || evtData['remarks'] || 'Yleinen UMM-ilmoitus'; if (titleText.length > 120) titleText = titleText.substring(0, 117) + '...'; processedEvents.push({ title: titleText, remarks: evtData['remarks'], start: eventStart, end: eventStop }); } } processedEvents.sort((a, b) => a.start - b.start); cachedEvents = processedEvents; isFirstLoad = false; lastUpdatedEl.innerText = 'Päivitetty: klo ' + now.toLocaleTimeString('fi-FI'); renderEvents(cachedEvents, false); }) .catch(e => { clearTimeout(timeoutId); if (!isFirstLoad) { renderEvents(cachedEvents, true); } else { ummContent.innerHTML = '
Virhe yhteydessä Nord Pooliin: ' + e.message + '
'; } }); }
